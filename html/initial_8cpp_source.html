348void node_initialsegment(int i, int j)
349{
350 //**********下面计算到相邻节点的长度和外法向**********//
351 point[i][j].weightnext.clear();
352 point[i][j].weightlast.clear();
353 point[i][j].segmentnext.clear();
354 point[i][j].segmentlast.clear();
355 point[i][j].anext.clear();
356 point[i][j].alast.clear();
357 point[i][j].Nlast.clear();
358 point[i][j].Nnext.clear();
359 point[i][j].nnext.clear();
360 point[i][j].nlast.clear();
361 point[i][j].flast.clear();
362 point[i][j].fnext.clear();
363 for (int r=0; r < point[i][j].neighbor_subcell.size(); r++)
364 {
365 int k, l;
366 k = point[i][j].neighbor_subcell[r] / m_point;
367 l = point[i][j].neighbor_subcell[r] % m_point;
368
369 int loc;
370 for (loc=0; loc<4; loc++)
371 {
372 if (subcell[k][l].vertex[loc] == point[i][j].q)
373 {
374 break;
375 }
376 }
377 //此时loc表示point(i,j)在*子网格*(k,l)中的局部编号
378
379 int noden, nodel;
380 if (loc == 3)
381 {
382 noden = 0;
383 }
384 else{
385 noden = loc + 1;
386 }
387
388 if (loc == 0)
389 {
390 nodel = 3;
391 }
392 else{
393 nodel = loc - 1;
394 }
395 //noder, nodel为(i,j)下一个和上一个局部坐标
396 point[i][j].segmentnext.push_back(subcell[k][l].vertex[noden]);
397 point[i][j].segmentlast.push_back(subcell[k][l].vertex[nodel]);
398
399 double * ftempnext = new double [2];
400 double * ftemplast = new double [2];
401 ftempnext[0] = 0;
402 ftempnext[1] = 0;
403 ftemplast[0] = r;
404 ftemplast[1] = 0;
405 point[i][j].fnext.push_back(ftempnext);
406 point[i][j].flast.push_back(ftemplast);
407
408 //下面计算加权外法向量
409 int inext, ilast, jnext, jlast;
410 inext = subcell[k][l].vertex[noden] / (m_point + 1);
411 jnext = subcell[k][l].vertex[noden] % (m_point + 1);
412 ilast = subcell[k][l].vertex[nodel] / (m_point + 1);
413 jlast = subcell[k][l].vertex[nodel] % (m_point + 1);
414
415 //下面存储每条边的权重
416 if (point[i][j].ifcenter == 1)
417 {
418 //是大网格中点
419 point[i][j].weightnext.push_back(1);
420 point[i][j].weightlast.push_back(1);
421 }
422 else if (point[i][j].ifedge == 0)
423 {
424 //是顶点

426 point[i][j].weightlast.push_back(1.0/6.0);
427 }
428 else{
429 //是边上的点
430 if (point[inext][jnext].ifcenter == 1)
431 {
432 point[i][j].weightnext.push_back(4.0/6.0);
433 }
434 else{
435 point[i][j].weightnext.push_back(2.0/6.0);
436 }
437
438 if (point[ilast][jlast].ifcenter == 1)
439 {
440 point[i][j].weightlast.push_back(4.0/6.0);
441 }
442 else{
443 point[i][j].weightlast.push_back(2.0/6.0);
444 }
445 }
446
447 //下面计算每条边单位外法向量
448 //先计算参考单元单位法向量
449 double * ntempnext = new double [2];
450 double * ntemplast = new double [2];
451
452 if (loc == 0)
453 {
454 //next
455 ntempnext[0] = 0;
456 ntempnext[1] = -1;
457 point[i][j].Nnext.push_back(ntempnext);
458 //last
459 ntemplast[0] = -1;
460 ntemplast[1] = 0;
461 point[i][j].Nlast.push_back(ntemplast);
462 }
463 else if(loc == 1)
464 {
465 //next
466 ntempnext[0] = 1;
467 ntempnext[1] = 0;
468 point[i][j].Nnext.push_back(ntempnext);
469 //last
470 ntemplast[0] = 0;
471 ntemplast[1] = -1;
472 point[i][j].Nlast.push_back(ntemplast);
473 }
474 else if(loc == 2)
475 {
476 //next
477 ntempnext[0] = 0;
478 ntempnext[1] = 1;
479 point[i][j].Nnext.push_back(ntempnext);
480 //last
481 ntemplast[0] = 1;
482 ntemplast[1] = 0;
483 point[i][j].Nlast.push_back(ntemplast);
484 }
485 else{
486 //next
487 ntempnext[0] = -1;
488 ntempnext[1] = 0;
489 point[i][j].Nnext.push_back(ntempnext);
490 //last
491 ntemplast[0] = 0;
492 ntemplast[1] = 1;
493 point[i][j].Nlast.push_back(ntemplast);
494 }
495
496 //下面计算物理空间单位外法向量(n = |J|J^{-T} N)
497 //并且计算加权外法向量(weight * s * n)
498 double detJ, xit, etat;
499 double ** J;
500 int Fcell, ifather, jfather, fatherloc;
501 Fcell = subcell[k][l].father_element;
502 ifather = Fcell / m_element;
503 jfather = Fcell % m_element;
504
505 //fatherloc记录在大网格中的局部编号
506 for (fatherloc=0; fatherloc<9; fatherloc++)
507 {
508 if(o[ifather][jfather].vertex[fatherloc] == point[i][j].q)
509 {
510 break;
511 }
512 }
513 xit = ref_xi[fatherloc];
514 etat = ref_eta[fatherloc];
515
516 J = o[ifather][jfather].getJacobiMatrix(xit,etat);
517
518 double norm;
519
520 double * antempnext = new double [2];
521 //next
522 antempnext[0] = J[1][1] * ntempnext[0] - J[1][0] * ntempnext[1];
523 antempnext[1] = - J[0][1] * ntempnext[0] + J[0][0] * ntempnext[1];
524 norm = antempnext[0] * antempnext[0] + antempnext[1] * antempnext[1];
525 norm = sqrt(norm);
526 antempnext[0] = antempnext[0] / norm;
527 antempnext[1] = antempnext[1] / norm;
528 //此时antempnext为物理空间外法向量
529 point[i][j].nnext.push_back(antempnext);
530 //计算所在边插值的|J|
531 double atnext;
532 detJ = o[ifather][jfather].edgedetJacobi(xit,etat,point[i][j].q,point[i][j].segmentnext[r]);
533 atnext = point[i][j].weightnext[r] * detJ * 2;
534 //此时atnext为物理空间上的加权边长，其中乘2是因为参考空间边长为2
535 point[i][j].anext.push_back(atnext);
536
537 double * antemplast = new double [2];
538 //last
539 antemplast[0] = J[1][1] * ntemplast[0] - J[1][0] * ntemplast[1];
540 antemplast[1] = - J[0][1] * ntemplast[0] + J[0][0] * ntemplast[1];
541 norm = antemplast[0] * antemplast[0] + antemplast[1] * antemplast[1];
542 norm = sqrt(norm);
543 antemplast[0] = antemplast[0] / norm;
544 antemplast[1] = antemplast[1] / norm;
545 //此时antemplast为物理空间上单位外法向量
546 point[i][j].nlast.push_back(antemplast);
547 //计算所在边插值的|J|
548 double atlast;
549 detJ = o[ifather][jfather].edgedetJacobi(xit,etat,point[i][j].q,point[i][j].segmentlast[r]);
550 atlast = point[i][j].weightlast[r] * detJ * 2;
551 //此时atlast为物理空间上加权边长
552 point[i][j].alast.push_back(atlast);
553
554 delete[] J[0];
555 delete[] J[1];
556 delete[] J;
557 }
558
559 return ;
560}

562double * mass_center(int i, int j)
563{
564 double * ce = new double [2];
565
566 int k;
567
568 double masst; //单元质量
569 double mf_xi, mf_eta; //质量矩
570
571 //********下面利用Gauss积分计算质量和质量矩************//
572 double rhot, jt; //记录密度和Jacobi行列式的点值
573 double xt, yt; //Gauss点的物理坐标
574 masst = 0;
575 mf_xi = 0;
576 mf_eta = 0;
577 for (k = 0; k < gpn; k++)
578 {
579 xt = o[i][j].phi_x0(Gausspoint_xi[k],Gausspoint_eta[k]);
580 yt = o[i][j].phi_y0(Gausspoint_xi[k],Gausspoint_eta[k]);
581
582 rhot = ini_rho(xt,yt);
583 jt = o[i][j].detJacobi_0(Gausspoint_xi[k],Gausspoint_eta[k]);
584
585 masst = masst + rhot * jt * Gaussweight[k];
586 mf_xi = mf_xi + rhot * Gausspoint_xi[k] * jt * Gaussweight[k];
587 mf_eta = mf_eta + rhot * Gausspoint_eta[k] * jt * Gaussweight[k];
588 }
589 o[i][j].mass = masst;
590
591 //**********下面计算质心坐标***********//
592
593 ce[0] = mf_xi / masst; //质心参考空间横坐标
594 ce[1] = mf_eta / masst; //质心参考空间纵坐标
595
596 //下面计算基函数345的质量平均
597 double a1, a2, a3;
598 a1 = 0;
599 a2 = 0;
600 a3 = 0;
601 for (k=0; k<gpn; k++)
602 {
603 xt = o[i][j].phi_x0(Gausspoint_xi[k],Gausspoint_eta[k]);
604 yt = o[i][j].phi_y0(Gausspoint_xi[k],Gausspoint_eta[k]);
605
606 rhot = ini_rho(xt,yt);
607 jt = o[i][j].detJacobi_0(Gausspoint_xi[k],Gausspoint_eta[k]);
608
609 a1 = a1 + rhot * (Gausspoint_xi[k] - ce[0])
610 * (Gausspoint_xi[k] - ce[0]) / 2.0 * jt * Gaussweight[k];
611 a2 = a2 + rhot * (Gausspoint_eta[k] - ce[0])
612 * (Gausspoint_eta[k] - ce[0]) / 2.0 * jt * Gaussweight[k];
613 a3 = a3 + rhot * (Gausspoint_xi[k] - ce[0])
614 * (Gausspoint_eta[k] - ce[0]) * jt * Gaussweight[k];
615 }
616 a1 = a1 / masst;
617 a2 = a2 / masst;
618 a3 = a3 / masst;
619
620 o[i][j].psi3avg = a1;
621 o[i][j].psi4avg = a2;
622 o[i][j].psi5avg = a3;
623
624 return ce;
625}

627double ** mass_matrix(int i, int j)
628{
629 int k, l;
630 double rhot, jt, xt, yt, temp;
631 double ** mma = new double * [pk];
632 for ( k = 0; k < pk; k++)
633 {
634 mma[k] = new double [pk];
635 }
636
637 for (k = 0; k<pk; k++)
638 {
639 for (l=0; l<pk; l++)
640 {
641 mma[k][l] = 0;
642 for (int r = 0; r <gpn; r++)
643 {
644 xt = o[i][j].phi_x0(Gausspoint_xi[r], Gausspoint_eta[r]);
645 yt = o[i][j].phi_y0(Gausspoint_xi[r], Gausspoint_eta[r]);
646
647 rhot = ini_rho(xt,yt);
648
649 jt = o[i][j].detJacobi_0(Gausspoint_xi[r], Gausspoint_eta[r]);
650
651 temp = o[i][j].Psi(k,Gausspoint_xi[r],Gausspoint_eta[r]);
652 temp = temp * o[i][j].Psi(l,Gausspoint_xi[r],Gausspoint_eta[r]);
653
654 mma[k][l] = mma[k][l] + rhot * temp * jt * Gaussweight[r];
655 }
656 }
657 }
658
659 return mma;
660}

722 xt = o[i][j].phi_x0(xit,etat);
723 yt = o[i][j].phi_y0(xit,etat);

726 jt =0;

728 {

734

737 else{

739 jt = 0;

741
742 mass = mass + rhot * jt * Gaussweight[s];
743 area = area + jt * Gaussweight[s];
744 }

747 subcell[k][l].area_s = area;
748 subcell[k][l].avg_den = mass / area;

869 + ini_rho_x(xt,yt) * x_xixi / (-rhot * rhot)

871 + (2 * ini_rho_y(xt,yt) * ini_rho_y(xt,yt) / (rhot*rhot*rhot) - ini_rho_yy(xt,yt) / (rhot*rhot)) * y_xi * y_xi
872 + ini_rho_y(xt,yt) * y_xixi / (-rhot * rhot);
873 o[i][j].nu[4] = (2 * ini_rho_x(xt,yt) * ini_rho_x(xt,yt) / (rhot*rhot*rhot) - ini_rho_xx(xt,yt) / (rhot*rhot)) * x_eta * x_eta
874 + (2 * ini_rho_x(xt,yt) * ini_rho_y(xt,yt) / (rhot*rhot*rhot) - ini_rho_xy(xt,yt) / (rhot*rhot)) * x_eta * y_eta
875 + ini_rho_x(xt,yt) * x_etaeta / (-rhot * rhot)
876 + (2 * ini_rho_x(xt,yt) * ini_rho_y(xt,yt) / (rhot*rhot*rhot) - ini_rho_xy(xt,yt) / (rhot*rhot)) * x_eta * y_eta
877 + (2 * ini_rho_y(xt,yt) * ini_rho_y(xt,yt) / (rhot*rhot*rhot) - ini_rho_yy(xt,yt) / (rhot*rhot)) * y_eta * y_eta
878 + ini_rho_y(xt,yt) * y_etaeta / (-rhot * rhot);
879 o[i][j].nu[5] = (2 * ini_rho_x(xt,yt) * ini_rho_x(xt,yt) / (rhot*rhot*rhot) - ini_rho_xx(xt,yt) / (rhot*rhot)) * x_xi * x_eta
880 + (2 * ini_rho_x(xt,yt) * ini_rho_y(xt,yt) / (rhot*rhot*rhot) - ini_rho_xy(xt,yt) / (rhot*rhot)) * x_eta * y_xi
881 + ini_rho_x(xt,yt) * x_xieta / (-rhot * rhot)

883 + (2 * ini_rho_y(xt,yt) * ini_rho_y(xt,yt) / (rhot*rhot*rhot) - ini_rho_yy(xt,yt) / (rhot*rhot)) * y_xi * y_eta
884 + ini_rho_y(xt,yt) * y_xieta / (-rhot * rhot);
885
886 o[i][j].ux[0] = ini_ux(xt,yt);
887 o[i][j].ux[1] = ini_ux_x(xt,yt) * x_xi + ini_ux_y(xt,yt) * y_xi;
888 o[i][j].ux[2] = ini_ux_x(xt,yt) * x_eta + ini_ux_y(xt,yt) * y_eta;
889 o[i][j].ux[3] = ini_ux_xx(xt,yt) * x_xi * x_xi + ini_ux_xy(xt,yt) * x_xi * y_xi
890 + ini_ux_x(xt,yt) * x_xixi + ini_ux_xy(xt,yt) * x_xi * y_xi

893 + ini_ux_x(xt,yt) * x_etaeta + ini_ux_xy(xt,yt) * x_eta * y_eta
894 + ini_ux_yy(xt,yt) * y_eta * y_eta + ini_ux_y(xt,yt) * y_etaeta;

896 + ini_ux_x(xt,yt) * x_xieta + ini_ux_xy(xt,yt) * x_xi * y_eta

900 o[i][j].uy[1] = ini_uy_x(xt,yt) * x_xi + ini_uy_y(xt,yt) * y_xi;

903 + ini_uy_x(xt,yt) * x_xixi + ini_uy_xy(xt,yt) * x_xi * y_xi
904 + ini_uy_yy(xt,yt) * y_xi * y_xi + ini_uy_y(xt,yt) * y_xixi;

906 + ini_uy_x(xt,yt) * x_etaeta + ini_uy_xy(xt,yt) * x_eta * y_eta
907 + ini_uy_yy(xt,yt) * y_eta * y_eta + ini_uy_y(xt,yt) * y_etaeta;
908 o[i][j].uy[5] = ini_uy_xx(xt,yt) * x_xi * x_eta + ini_uy_xy(xt,yt) * x_eta * y_xi
909 + ini_uy_x(xt,yt) * x_xieta + ini_uy_xy(xt,yt) * x_xi * y_eta
910 + ini_uy_yy(xt,yt) * y_xi * y_eta + ini_uy_y(xt,yt) * y_xieta;
911
912 double uxt, uyt, gammat;
913 uxt = ini_ux(xt,yt);
914 uyt = ini_uy(xt,yt);
915 gammat = o[i][j].gamma;
916 taux = ini_p_x(xt,yt) / ((gammat - 1) * rhot)
917 - ini_p(xt,yt) * ini_rho_x(xt,yt) / ((gammat - 1) * rhot * rhot)
918 + uxt * ini_ux_x(xt,yt) + uyt * ini_uy_x(xt,yt);
919 tauy = ini_p_y(xt,yt) / ((gammat - 1) * rhot)
920 - ini_p(xt,yt) * ini_rho_y(xt,yt) / ((gammat - 1) * rhot * rhot)
921 + uxt * ini_ux_y(xt,yt) + uyt * ini_uy_y(xt,yt);
922
923 tauxx = ini_ux_x(xt,yt) * ini_ux_x(xt,yt) + uxt * ini_ux_xx(xt,yt)
924 + ini_uy_x(xt,yt) * ini_uy_x(xt,yt) + uyt * ini_uy_xx(xt,yt)
925 + ini_p_xx(xt,yt) / ((gammat - 1) * rhot) - ini_p_x(xt,yt) * ini_rho_x(xt,yt) / ((gammat - 1) * rhot * rhot)
926 - (ini_p_x(xt,yt) * ini_rho_x(xt,yt) + ini_p(xt,yt) * ini_rho_xx(xt,yt)) / ((gammat - 1) * rhot * rhot)
927 + ini_p(xt,yt) * ini_rho_x(xt,yt) * 2 * rhot * ini_rho_x(xt,yt) / ((gammat - 1) * pow(rhot,4));
928
929 tauyy = ini_ux_y(xt,yt) * ini_ux_y(xt,yt) + uxt * ini_ux_yy(xt,yt)
930 + ini_uy_y(xt,yt) * ini_uy_y(xt,yt) + uyt * ini_uy_yy(xt,yt)
931 + ini_p_yy(xt,yt) / ((gammat - 1) * rhot) - ini_p_y(xt,yt) * ini_rho_y(xt,yt) / ((gammat - 1) * rhot * rhot)
932 - (ini_p_y(xt,yt) * ini_rho_y(xt,yt) + ini_p(xt,yt) * ini_rho_yy(xt,yt)) / ((gammat - 1) * rhot * rhot)
933 + ini_p(xt,yt) * ini_rho_y(xt,yt) * 2 * rhot * ini_rho_y(xt,yt) / ((gammat - 1) * pow(rhot,4));
934
935 tauxy = ini_ux_x(xt,yt) * ini_ux_y(xt,yt) + uxt * ini_ux_xy(xt,yt)
936 + ini_uy_x(xt,yt) * ini_uy_y(xt,yt) + uyt * ini_uy_xy(xt,yt)
937 + ini_p_xy(xt,yt) / ((gammat - 1) * rhot) - ini_p_x(xt,yt) * ini_rho_y(xt,yt) / ((gammat - 1) * rhot * rhot)
938 - (ini_p_y(xt,yt) * ini_rho_x(xt,yt) + ini_p(xt,yt) * ini_rho_xy(xt,yt)) / ((gammat - 1) * rhot * rhot)
939 + ini_p(xt,yt) * ini_rho_x(xt,yt) * 2 * rhot * ini_rho_y(xt,yt) / ((gammat - 1) * pow(rhot,4));
940
941 o[i][j].tau[0] = ini_p(xt,yt) / ((gammat - 1) * rhot)
942 + 0.5 * (uxt * uxt + uyt * uyt);
943 o[i][j].tau[1] = taux * x_xi + tauy * y_xi;
944 o[i][j].tau[2] = taux * x_eta + tauy * y_eta;
945 o[i][j].tau[3] = tauxx * x_xi * x_xi + tauxy * x_xi * y_xi
946 + taux * x_xixi + tauxy * x_xi * y_xi
947 + tauyy * y_xi * y_xi + tauy * y_xixi;
948
949 o[i][j].tau[4] = tauxx * x_eta * x_eta + tauxy * x_eta * y_eta
950 + taux * x_etaeta + tauxy * x_eta * y_eta
951 + tauyy * y_eta * y_eta + tauy * y_etaeta;
952
953 o[i][j].tau[5] = tauxx * x_xi * x_eta + tauxy * x_eta * y_xi
954 + taux * x_xieta + tauxy * x_xi * y_eta
955 + tauyy * y_xi * y_eta + tauy * y_xieta;
956
957 o[i][j].rho_0 = rhot;
958
959 o[i][j].c_center = gammat * ini_p(xt,yt) / rhot;
960 o[i][j].c_center = sqrt(o[i][j].c_center);
961 }
962 }
963
964 return ;
