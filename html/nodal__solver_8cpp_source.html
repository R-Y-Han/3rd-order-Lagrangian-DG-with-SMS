67double density_SMS(int i, int j)
68{
69 double rhoc;
70
71 //注意i,j是子网格的编号
72
73 double rho_s, rho_nus;
74 rho_s = subcell[i][j].avg_den;
75
76 //k,l是大网格的编号
77 int k, l, r, s;
78 k = subcell[i][j].father_element / m_element;
79 l = subcell[i][j].father_element % m_element;
80
81 //下面找到子网格的区域
82 double Al, Ar, Bb, Bt;
83 for (r=0; r<4; r++)
84 {
85 if (o[k][l].subcell[r] == subcell[i][j].q)
86 {
87 break;
88 }
89 }
90 switch(r)
91 {
92 case 0:
93 {
94 Al = -1;
95 Ar = 0;
96 Bb = -1;
97 Bt = 0;
98 break;
99 }
100 case 1:
101 {
102 Al = 0;
103 Ar = 1;
104 Bb = -1;
105 Bt = 0;
106 break;
107 }
108 case 2:
109 {
110 Al = 0;
111 Ar = 1;
112 Bb = 0;
113 Bt = 1;
114 break;
115 }
116 case 3:
117 {
118 Al = -1;
119 Ar = 0;
120 Bb = 0;
121 Bt = 1;
122 break;
123 }
124 default:
125 {
126 Al = 0;
127 Ar = 0;
128 Bb = 0;
129 Bt = 0;
130 break;
131 }
132 }
133
134 //下面计算大网格的质量平均
135 double mt, xit, etat, rhot, jt;
136 mt=0;
137 for (s=0; s<gpn; s++)
138 {
139 xit = Gausspoint_xi[s];
140 etat = Gausspoint_eta[s];
141
142 rhot = 0;
143 jt =0;
144 if (xit >= Al && xit <= Ar)
145 {
146 if (etat >= Bb && etat <= Bt)
147 {
148 //在子网格中
149 rhot = 0;
150 for (int tt=0; tt<pk; tt++)
151 {
152 rhot = rhot + o[k][l].nu[tt] * o[k][l].Psi(tt,xit,etat);
153 }
154 rhot = 1.0 / rhot;
155 jt = o[k][l].detJacobi(xit,etat);
156 }
157 }
158 mt = mt + rhot * jt * Gaussweight[s];
159 }
160
161 rho_nus = mt / subcell[i][j].area_s;
162
163 rhoc = (rho_s - rho_nus);
164
165 return rhoc;
166}

472void BCnormal_velocity(int i, int j, double vnt)
473{
474 double ux, uy, Pre;
475 double nstarx1, nstary1, nstarx2, nstary2;
476 int r, node1, node2, k1, l1, k2, l2;
477
478 //先找到第一个相邻边界点的外法向
479 for (r=0; r<point[i][j].neighbor_subcell.size(); r++)
480 {
481 node1 = point[i][j].segmentlast[r];
482 k1 = node1 / (m_point + 1);
483 l1 = node1 % (m_point + 1);
484 if (point[k1][l1].boundary == 1)
485 {
486 nstarx1 = point[i][j].nlast[r][0];
487 nstary1 = point[i][j].nlast[r][1];
488 break;
489 }
490 node1 = point[i][j].segmentnext[r];
491 k1 = node1 / (m_point + 1);
492 l1 = node1 % (m_point + 1);
493 if (point[k1][l1].boundary == 1)
494 {
495 nstarx1 = point[i][j].nnext[r][0];
496 nstary1 = point[i][j].nnext[r][1];
497 break;
498 }
499 }
500 //此时code1是第一个相邻边界点，nstarx1, nstary1是第一个边界外法向量
501
502 //再找第二个边界点
503 for (r=0; r<point[i][j].neighbor_subcell.size(); r++)
504 {
505 node2 = point[i][j].segmentlast[r];
506 k2 = node2 / (m_point + 1);
507 l2 = node2 % (m_point + 1);
508 if (point[k2][l2].boundary == 1)
509 {
510 if (node2 != node1)
511 {
512 nstarx2 = point[i][j].nlast[r][0];
513 nstary2 = point[i][j].nlast[r][1];
514 break;
515 }
516 }
517 node2 = point[i][j].segmentnext[r];
518 k2 = node2 / (m_point + 1);
519 l2 = node2 % (m_point + 1);
520 if (point[k2][l2].boundary == 1)
521 {
522 if (node2 != node1)
523 {
524 nstarx2 = point[i][j].nnext[r][0];
525 nstary2 = point[i][j].nnext[r][1];
526 break;
527 }
528 }
529 }
530
531 //判断两个外法向是否平行
532 double para;
533 para = nstarx1 * nstary2 - nstary1 * nstarx2;
534 para = abs(para);
535
536 if (para >= 1e-6)
537 {
538 //不平行
539 point[i][j].upstarx = vnt * (nstary2 - nstary1) / para;
540 point[i][j].upstary = vnt * (- nstarx2 + nstarx1) / para;
541 return ;
542 }
543
544 //平行
545 double nx, ny;
546 nx = 0.5 * (nstarx1 + nstarx2);
547 ny = 0.5 * (nstary1 + nstary2);
548
549 Pre = ux * nx + uy * ny;
550
551 point[i][j].upstarx = ux + Pre * nx + vnt;
552 point[i][j].upstary = uy + Pre * ny + vnt;
553
554 return ;
555}

557void BCpressure(int i, int j, double pcorlast, double pcornext)
558{
559 double ux, uy;

645 + pcornext * nstarxn;
646 point[i][j].upstary = uy + pcorlast * nstaryl
