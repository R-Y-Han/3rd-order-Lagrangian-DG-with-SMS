20double * RK_step1(double * vn, double * Rn,
21 double ** Minv, double delta_t, int len)
22{
23 double * vs1 = new double [len];
24 double * temp = new double [len];
25 int i, j;
26 for (i=0; i<len; i++)
27 {
28 vs1[i] = 0;
29 temp[i] = 0;
30 }
31
32 for (i=0; i<len; i++)
33 {
34 for (j=0; j<len; j++)
35 {
36 temp[i] = temp[i] + Minv[i][j] * Rn[j];
37 }
38 }
39
40 for (i=0; i<len; i++)
41 {
42 vs1[i] = vn[i] + delta_t * temp[i];
43 }
44
45 delete[] temp;
46 return vs1;
47}

49double * RK_step2(double * vn, double * vs1, double * Rs1,
50 double ** Minv, double delta_t, int len)
51{
52 double * vs2 = new double [len];
53 double * temp = new double [len];
54 int i, j;
55 for (i=0; i<len; i++)
56 {
57 vs2[i] = 0;
58 temp[i] = 0;
59 }
60
61 for (i=0; i<len; i++)
62 {
63 for (j=0; j<len; j++)
64 {
65 temp[i] = temp[i] + Minv[i][j] * Rs1[j];
66 }
67 }
68
69 for (i=0; i<len; i++)
70 {
71 vs2[i] = 3.0 / 4.0 * vn[i] + 1.0 / 4.0 * vs1[i]
72 + 1.0 / 4.0 * delta_t * temp[i];
73 }
74
75 delete[] temp;
76 return vs2;
77}

79double * RK_step3(double * vn, double * vs2, double * Rs2,
80 double ** Minv, double delta_t, int len)
81{
82 double * vs3 = new double [len];
83 double * temp = new double [len];
84 int i, j;
85 for (i=0; i<len; i++)
86 {
87 vs3[i] = 0;
88 temp[i] = 0;
89 }
90
91 for (i=0; i<len; i++)
92 {
93 for (j=0; j<len; j++)
94 {
95 temp[i] = temp[i] + Minv[i][j] * Rs2[j];
96 }
97 }
98
99 for (i=0; i<len; i++)
100 {
101 vs3[i] = 1.0 / 3.0 * vn[i] + 2.0 / 3.0 * vs2[i]
102 + 2.0 / 3.0 * delta_t * temp[i];
103 }
104
105 delete[] temp;
106 return vs3;

566void update_geo()
567{
568 int fk, fl, sk, sl, pi, pj ,r, s;
569
570 //下面更新网格顶点位置
571 for(fk=0; fk<n_element; fk++)
572 {
573 for (fl=0; fl<m_element; fl++)
574 {
575 for (r=0; r<9; r++)
576 {
577 pi = o[fk][fl].vertex[r] / (m_point + 1);
578 pj = o[fk][fl].vertex[r] % (m_point + 1);
579 o[fk][fl].vx[r] = point[pi][pj].x;
580 o[fk][fl].vy[r] = point[pi][pj].y;
581 }
582 }
583 }
584
585 //下面更新子网格的面积
586 for (sk=0; sk<n_point; sk++)
587 {
588 for (sl=0; sl<m_point; sl++)
589 {
590 double area, jt, xit, etat;
591 double Al, Ar, Bb, Bt;
592 //找到子网格在大网格中的位置
593 fk = subcell[sk][sl].father_element / m_element;
594 fl = subcell[sk][sl].father_element % m_element;
595 for (r=0; r<4; r++)
596 {
597 if (o[fk][fl].subcell[r] == subcell[sk][sl].q)
598 {
599 break;
600 }
601 }
602 //找到子网格对应拉氏区域
603 switch(r)
604 {
605 case 0:
606 {
607 Al = -1;
608 Ar = 0;
609 Bb = -1;
610 Bt = 0;
611 break;
612 }
613 case 1:
614 {
615 Al = 0;
616 Ar = 1;
617 Bb = -1;
618 Bt = 0;
619 break;
620 }
621 case 2:
622 {
623 Al = 0;
624 Ar = 1;
625 Bb = 0;
626 Bt = 1;
627 break;
628 }
629 case 3:
630 {
631 Al = -1;
632 Ar = 0;
633 Bb = 0;
634 Bt = 1;
635 break;
636 }
637 default:
638 {
639 Al = 0;
640 Ar = 0;
641 Bb = 0;
642 Bt = 0;
643 break;
644 }
645 }
646 //计算面积
647 area = 0;
648 for (s=0; s<gpn; s++)
649 {
650 xit = Gausspoint_xi[s];
651 etat = Gausspoint_eta[s];
652 if (xit >= Al && xit <=Ar)
653 {
654 if (etat >= Bb && etat <= Bt)
655 {
656 //在子网格中
657 jt = o[fk][fl].detJacobi(xit,etat);
658 }
659 }
660 else{
661 jt = 0;
662 }
663 area = area + jt * Gaussweight[s];
664 }
665 subcell[sk][sl].area_s = area;
666 }
667 }
668
669 //下面更新节点单位外法向量和加权边长
670 for (pi=0; pi<=n_point; pi++)
671 {
672 for (pj=0; pj<=m_point; pj++)
673 {
674 for (r=0; r<point[pi][pj].neighbor_subcell.size(); r++)
675 {
676 sk = point[pi][pj].neighbor_subcell[r] / m_point;
677 sl = point[pi][pj].neighbor_subcell[r] % m_point;
678
679 fk = subcell[sk][sl].father_element / m_element;
680 fl = subcell[sk][sl].father_element % m_element;
681
682 //找到节点在大网格中的位置
683 int loc;
684 for (loc=0; loc<9; loc++)
685 {
686 if (o[fk][fl].vertex[loc] == point[pi][pj].q)
687 {
688 break;
689 }
690 }
691 double xit, etat;
692 xit = ref_xi[loc];
693 etat = ref_eta[loc];
694
695 //下面计算物理空间单位外法向量
696 double * nl = new double [2];
697 double * nn = new double [2];
698 double ** J;
699 double norm;
700
701 J = o[fk][fl].getJacobiMatrix(xit,etat);
702 //last
703 nl[0] = J[1][1] * point[pi][pj].Nlast[r][0]
704 - J[1][0] * point[pi][pj].Nlast[r][1];
705 nl[1] = - J[0][1] * point[pi][pj].Nlast[r][0]
706 + J[0][0] * point[pi][pj].Nlast[r][1];
707 norm = nl[0] * nl[0] + nl[1] * nl[1];
708 norm = sqrt(norm);
709 nl[0] = nl[0] / norm;
710 nl[1] = nl[1] / norm;
711 point[pi][pj].nlast[r][0] = nl[0];
712 point[pi][pj].nlast[r][1] = nl[1];
713
714 //next
715 nn[0] = J[1][1] * point[pi][pj].Nnext[r][0]
716 - J[1][0] * point[pi][pj].Nnext[r][1];
717 nn[1] = - J[0][1] * point[pi][pj].Nnext[r][0]
718 + J[0][0] * point[pi][pj].Nnext[r][1];
719 norm = nn[0] * nn[0] + nn[1] * nn[1];
720 norm = sqrt(norm);
721 nn[0] = nn[0] / norm;
722 nn[1] = nn[1] / norm;
723 point[pi][pj].nnext[r][0] = nn[0];
724 point[pi][pj].nnext[r][1] = nn[1];
725
726 //下面计算加权边长
727 double al, an, detJ;
728 //last
729 detJ = o[fk][fl].edgedetJacobi(xit,etat,
730 point[pi][pj].q,point[pi][pj].segmentlast[r]);
731 al = point[pi][pj].weightlast[r] * detJ * 2;
732 point[pi][pj].alast[r] = al;
733
734 //next
735 detJ = o[fk][fl].edgedetJacobi(xit,etat,
736 point[pi][pj].q,point[pi][pj].segmentnext[r]);
737 an = point[pi][pj].weightnext[r] * detJ * 2;
738 point[pi][pj].anext[r] = an;
739
740 delete[] J[0];
741 delete[] J[1];
742 delete[] J;
743 delete[] nl;
744 delete[] nn;
745 }
746 }
747 }
748
749 return ;
750}

752double choosedt(double dtn)
753{
754 double dt, ac, temp;
755 int sk, sl, fk, fl, pi, pj, pk, pl, r, s;
756 double ax, ay, bx, by, len;
757
758 //下面计算每个子网格中最短边长
759 //反正都是近似取值，两点间距就直接用直线计算
760 dt = 1000;
761 for (sk=0; sk<n_point; sk++)
762 {
763 for (sl=0; sl<m_point; sl++)
764 {
765 //先记录声速
766 fk = subcell[sk][sl].father_element / m_element;
767 fl = subcell[sk][sl].father_element % m_element;
768 ac = o[fk][fl].c_center;
769
770 //对每个子网格的顶点循环
771 for (r=0; r<4; r++)
772 {
773 pi = subcell[sk][sl].vertex[r] / (m_point + 1);
774 pj = subcell[sk][sl].vertex[r] % (m_point + 1);
775
776 ax = point[pi][pj].x;
777 ay = point[pi][pj].y;
778
779 //下面计算到其他节点的时间
780 for (s=r+1; s<4; s++)
781 {
782 pk = subcell[sk][sl].vertex[s] / (m_point+1);
783 pl = subcell[sk][sl].vertex[s] % (m_point+1);
784
785 bx = point[pk][pl].x;
786 by = point[pk][pl].y;
787
788 //下面计算到相邻点的直线距离
789 len = (bx - ax) * (bx - ax) + (by - ay) * (by - ay);
790 len = sqrt(len);
791
792 //下面判断是否为最短时间
793 temp = len / ac;
794 if (temp < dt)
795 {
796 dt = temp;
797 }
798 }
799 }
800 }
801 }
802 dt = C_cfl * dt;
803 dt = min(dt,C_m * dtn);
804
805 return dt;
806}
