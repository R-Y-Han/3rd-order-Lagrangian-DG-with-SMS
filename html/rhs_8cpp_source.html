63double * specific_volume_matrix(int i, int j)
64{
65 double * R = new double [pk];
66 int pi, pj, sk, sl, q, r, s, ba;
67 double xit, etat;
68
69 for (r=0; r<pk; r++)
70 {
71 R[r] = 0;
72 }
73
74 //下面计算边界通量
75 for (r=0; r<4; r++)
76 {
77 //在每个子网格中计算
78 sk = o[i][j].subcell[r] / m_point;
79 sl = o[i][j].subcell[r] % m_point;
80
81 for (s=0; s<4; s++)
82 {
83 //在每个子网格的顶点上计算
84 pi = subcell[sk][sl].vertex[s] / (m_point + 1);
85 pj = subcell[sk][sl].vertex[s] % (m_point + 1);
86 //中心点不用计算
87 if (point[pi][pj].ifcenter == 1)
88 {
89 continue;
90 }
91
92 //找出所求节点在大网格中对应的局部拉氏坐标
93 double * co;
94 co = get_xieta(point[pi][pj].q,o[i][j].q);
95 xit = co[0];
96 etat = co[1];
97 delete[] co;
98
99 //找出子网格在节点相邻网格的编号
100 q = get_subcell_in_vertexneighbor(subcell[sk][sl].q,point[pi][pj].q);
101
102 for (ba=0; ba<pk; ba++)
103 {
104 double tempx, tempy;
105 tempx = point[pi][pj].alast[q] * point[pi][pj].nlast[q][0]
106 + point[pi][pj].anext[q] * point[pi][pj].nnext[q][0];
107
108 tempy = point[pi][pj].alast[q] * point[pi][pj].nlast[q][1]
109 + point[pi][pj].anext[q] * point[pi][pj].nnext[q][1];
110
111 R[ba] = R[ba] + o[i][j].Psi(ba,xit,etat)
112 * (tempx * point[pi][pj].upstarx
113 + tempy * point[pi][pj].upstary);
114 }
115 }
116 }
117
118 //下面计算体积分
119 for (r=0; r<gpn; r++)
120 {
121 xit = Gausspoint_xi[r];
122 etat = Gausspoint_eta[r];
123
124 //下面计算在Gauss求积点的速度
125 double ux, uy;
126 ux = 0;
127 uy = 0;
128 for (s=0; s<pk; s++)
129 {
130 ux = ux + o[i][j].ux[s] * o[i][j].Psi(s,xit,etat);
131 uy = uy + o[i][j].uy[s] * o[i][j].Psi(s,xit,etat);
132 }
133
134 //下面计算Gauss求积点处的Jacobi矩阵
135 double ** J = o[i][j].getJacobiMatrix(xit,etat);
136
137 //下面计算R的第ba个分量
138 for (ba=0; ba<pk; ba++)
139 {
140 double psi_xi, psi_eta;
141 psi_xi = o[i][j].Psi_xi(ba,xit,etat);
142 psi_eta = o[i][j].Psi_eta(ba,xit,etat);
143
144 R[ba] = R[ba] - Gaussweight[r] *
145 ( ux * (J[1][1] * psi_xi - J[1][0] * psi_eta)
146 + uy * (-J[0][1] * psi_xi + J[0][0] * psi_eta));
147 }
148//cout<<ux + o[i][j].phi_x0(xit,etat)<<endl;
149 delete[] J[0];
150 delete[] J[1];
151 delete[] J;
152 }
153
154 return R;
155}

157double ** velocity_matrix(int i, int j)
158{
159 double ** R = new double * [pk];
160 int pi, pj, sk, sl, q, r, s, ba;
161 double xit, etat;
162 for (r=0; r<pk; r++)
163 {
164 R[r] = new double [2];
165 }
166 for (r=0; r<pk; r++)
167 {
168 for (s=0; s<2; s++)
169 {
170 R[r][s] = 0;
171 }
172 }
173
174 //下面计算边界通量
175 for (r=0; r<4; r++)
176 {
177 //对每个子网格计算
178 sk = o[i][j].subcell[r] / m_point;
179 sl = o[i][j].subcell[r] % m_point;
180
181 for (s=0; s<4; s++)
182 {
183 //在每个子网格的节点计算
184 pi = subcell[sk][sl].vertex[s] / (m_point + 1);
185 pj = subcell[sk][sl].vertex[s] % (m_point + 1);
186 //中心点不用计算
187 if (point[pi][pj].ifcenter == 1)
188 {
189 continue;
190 }
191
192 //找出所求节点在大网格中对应的局部坐标
193 double * co;
194 co = get_xieta(point[pi][pj].q,o[i][j].q);
195 xit = co[0];
196 etat = co[1];
197 delete[] co;
198
199 //找出子网格在节点相邻网格中的编号
200 q = get_subcell_in_vertexneighbor(subcell[sk][sl].q,point[pi][pj].q);
201
202 for (ba=0; ba<pk; ba++)
203 {
204 R[ba][0] = R[ba][0] + o[i][j].Psi(ba,xit,etat)
205 * (point[pi][pj].flast[q][0]
206 + point[pi][pj].fnext[q][0]);
207 R[ba][1] = R[ba][1] + o[i][j].Psi(ba,xit,etat)
208 * (point[pi][pj].flast[q][1]
209 + point[pi][pj].fnext[q][1]);
210 }
211 }
212 }
213
214 //下面计算体积分
215 for (r=0; r<gpn; r++)
216 {
217 xit = Gausspoint_xi[r];
218 etat = Gausspoint_eta[r];
219
220 //下面计算Gauss求积点的压力
221 double rhot, ux, uy, e, sigma;
222 rhot = 0;
223 ux = 0;
224 uy = 0;
225 e = 0;
226 for (s=0; s<pk; s++)
227 {
228 rhot = rhot + o[i][j].nu[s] * o[i][j].Psi(s,xit,etat);
229 ux = ux + o[i][j].ux[s] * o[i][j].Psi(s,xit,etat);
230 uy = uy + o[i][j].uy[s] * o[i][j].Psi(s,xit,etat);
231 e = e + o[i][j].tau[s] * o[i][j].Psi(s,xit,etat);
232 }
233 rhot = 1.0 / rhot;
234 e = e - 0.5 * (ux * ux + uy * uy);
235 e = max(1e-9,e);
236 sigma = - EOS(o[i][j].gamma,rhot,e);
237
238 //下面计算Gauss求积点处Jacobi矩阵
239 double ** J = o[i][j].getJacobiMatrix(xit,etat);
240
241 //下面计算R的第ba个分量
242 for (ba=0; ba<pk; ba++)
243 {
244 double psi_xi, psi_eta;
245 psi_xi = o[i][j].Psi_xi(ba,xit,etat);
246 psi_eta = o[i][j].Psi_eta(ba,xit,etat);
247
248 R[ba][0] = R[ba][0] - Gaussweight[r] * sigma *
249 (J[1][1] * psi_xi - J[1][0] * psi_eta);
250 R[ba][1] = R[ba][1] - Gaussweight[r] * sigma *
251 (-J[0][1] * psi_xi + J[0][0] * psi_eta);
252 }
253
254 delete[] J[0];
255 delete[] J[1];
256 delete[] J;
257 }
258
259 double ** Rt = new double * [2];
260 Rt[0] = new double [pk];
261 Rt[1] = new double [pk];
262 for (r=0; r<pk; r++)
263 {
264 Rt[0][r] = R[r][0];
265 Rt[1][r] = R[r][1];
266 }
267
268 for (r = 0; r<pk; r++)
269 {
270 delete[] R[r];
271 }
272 delete[] R;
273
274 return Rt;
